149 void Fish::grow(double tsb, double temp){
150  if (par.growth_model == GrowthModel::Dankel22){
151  // do nothing. age is incremented by population update
152  gsi_effective = par.gsi; // required if new fecundity model is used in combination with old growth model
153  }
154  else if (par.growth_model == GrowthModel::Bioenergetic){
155  double tsb_ano = tsb - par.tsbmean;
156  double temp_ano = temp - par.Tmean;
157 
158  // This is generalized increment l2^y1y2 - l1^y1y2
159  double dl = fish::dl_power(tsb_ano, temp_ano, par.gamma1, par.gamma2, par.alpha1, par.alpha2, par.beta1, par.beta2);
160  double dl_pot = fish::dl_power(-par.tsbmean, temp_ano, par.gamma1, par.gamma2, par.alpha1, par.alpha2, par.beta1, par.beta2);
161 
162  double lnew, lnew_pot;
163  if (isMature){
164  lnew = fish::length_adult(length, dl, par.gamma1, par.gamma2, par.gsi);
165  lnew_pot = fish::length_adult(length, dl_pot, par.gamma1, par.gamma2, par.gsi);
166  }
167  else{
168  lnew = fish::length_juvenile(length, dl, par.gamma1, par.gamma2);
169  lnew_pot = fish::length_juvenile(length, dl_pot, par.gamma1, par.gamma2);
170  }
171 
172  // ------ This is linear increment, just for analysis --------
173  dl_real = lnew - length;
174  dl_potential = lnew_pot - length;
175  //cout << "tsb_ano = " << tsb << " / " << par.tsbmean << ", fac = " << dl_real << " / " << dl_potential << endl;
176  // -----------------------------------------------------------
177 
178  gsi_effective = fish::gsi(lnew, length, dl, par.gamma1, par.gamma2, par.alpha1, par.alpha2);
179  set_length(lnew);
180  }
181  else{
182  throw std::runtime_error("Invalid growth model specified");
183  }

195 double Fish::produceRecruits(double ssb, double temp){
196  double temp_ano = temp - par.Tref;
197  if (par.recruitment_model == RecruitmentModel::BevertonHoltDirect){
198  double recruits = par.r0 * weight * 1 / (1 + ssb/par.Bhalf);
199  return recruits;
200  }
201  else if (par.recruitment_model == RecruitmentModel::RickerDirect){
202  double recruits = par.r0 * weight * exp(par.beta4*temp_ano) * pow(2, -ssb/par.Bhalf);
203  return recruits;
204  }
205  else if (par.recruitment_model == RecruitmentModel::BevertonHoltBioenergetic){
206  double eggs = fish::fecundity(weight, par.delta, gsi_effective); // Total eggs produced
207  double recruits = eggs * par.s0 * 1 / (1 + ssb/par.Bhalf); // offspring surviving density-dependent recruitment
208  return recruits;
209  }
210  else if (par.recruitment_model == RecruitmentModel::RickerBioenergetic){
211  double eggs = fish::fecundity(weight, par.delta, gsi_effective); // Total eggs produced
212  double recruits = eggs * par.s0 * exp(par.beta4*temp_ano) * pow(2, -ssb/par.Bhalf); // offspring surviving density-dependent recruitment
213  return recruits;
214  }
